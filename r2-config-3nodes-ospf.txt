!!! CISCO CSR1000v r2 configuration !!!

! Interfaces configuration

! GigabitEthernet2 (c2 <-> r2)
interface GigabitEthernet2
 ip address 10.0.2.1 255.255.255.0
 no shutdown 
!

! GigabitEthernet3 (r2 <-> r1)
interface GigabitEthernet3
 ip address 192.168.10.2 255.255.255.252
 no shutdown
!

! GigabitEthernet3 (r2 <-> r3)
interface GigabitEthernet4
 ip address 192.168.30.1 255.255.255.252
 no shutdown
!

! OSPF routes
router ospf 1
network 10.0.2.0 0.0.0.255 area 1
network 192.168.10.0 0.0.0.3 area 1
network 192.168.30.0 0.0.0.3 area 1
!

!!! END CONFIG !!!
